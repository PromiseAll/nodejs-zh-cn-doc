新增于: v18.6.0
  • moduleName <string> 模块名称
  • 返回: <boolean> 如果模块是内置的，则返回 true，否则返回 false
import { isBuiltin } from 'node:module';
isBuiltin('node:fs'); // true
isBuiltin('fs'); // true
isBuiltin('wss'); // false
